    /** @module gains */
import constants from "./constants.js";
import csvIO from "./csv/io.js";

const { CSV_HEADERS: H, TYPE, XTZ } = constants;

const makeFifoLedger = (asset) => {
  return {
    asset,
    amounts: [],
    dispossesed: [],
  };
};

const add = (ledger, amount, fiat, date) => {
  ledger.amounts.push({
    amount,
    fiat,
    date,
  });
  ledger.amounts = ledger.amounts.sort((a, b) => a.amount - b.amount);
  return ledger;
};

const remove = (ledger, amount, fiat, date, type) => {
  let remaining = amount;
  while (remaining > Number.EPSILON) {
    const top = ledger.amounts.splice(0, 1)[0];
    if (top === undefined) {
      console.log("undefined", ledger, amount, remaining);
      break;
    }
    let subamount;
    if (top.amount >= remaining) {
      subamount = remaining;
      top.amount = top.amount - subamount;
      ledger.amounts.splice(0, 0, top);
    } else {
      subamount = remaining - top.amount;
    }
    const proceeds = subamount * fiat;
    const basis = subamount * top.fiat;
    const gains = proceeds - basis;
    ledger.dispossesed.push({
      type,
      asset: ledger.asset,
      amount: subamount,
      acquired: top.date,
      basis,
      dispossesed: date,
      proceeds,
      gains,
      days: (Date.parse(date) - Date.parse(top.date)) / (1000 * 3600 * 24),
    });
    remaining -= subamount;
  }
};

/**
 * Generates gains and losses report
 *
 * @param {Array} collatedRows - Rows from {@link module:collation~collate}
 *
 * @returns {array} rows of gains and losses
 * */
const generateReport = async (year, collatedRows) => {
  const ledgers = {};
  const makeTokenKey = (token, tokenId) => {
    if (tokenId === "") {
      return token;
    } else {
      return `${token}-${tokenId}`;
    }
  };
  for (const row of collatedRows) {
    const t = row[H.TYPE];
    const date = row[H.TIMESTAMP];
    const i = row[H.IN_TOKEN];
    const iId = row[H.IN_TOKEN_ID];
    const o = row[H.OUT_TOKEN];
    const oId = row[H.OUT_TOKEN_ID];
    const fiat = row[H.FIAT];
    const iKey = makeTokenKey(i, iId);
    const oKey = makeTokenKey(o, oId);

    // xtz or token in
    if (
      t === TYPE.RECEIVE ||
      t === TYPE.RECEIVE_TOKEN ||
      t === TYPE.TRADE_FIAT_OUT ||
      t === TYPE.SALE ||
      t === TYPE.SALE_RESALE ||
      t === TYPE.AIRDROP
    ) {
      if (i !== "" && ledgers[iKey] === undefined) {
        ledgers[iKey] = makeFifoLedger(iKey);
      }
      let inFiat = 0;
      const inIsXTZ = i === XTZ;
      if (inIsXTZ) {
        inFiat = fiat;
      }
      add(ledgers[iKey], parseFloat(row[H.IN_AMT]), inFiat, date);

      // token out
      if (t === TYPE.SALE_RESALE) {
        if (o !== "" && ledgers[oKey] === undefined) {
          ledgers[oKey] = makeFifoLedger(oKey);
        }
        let inAmt = parseFloat(row[H.IN_AMT]);
        let outAmt = parseFloat(row[H.OUT_AMT]);
        const outFiat = (fiat * inAmt) / outAmt;
        remove(ledgers[oKey], outAmt, outFiat, date, t);
      }
    }

    // xtz out
    if (t === TYPE.TRADE_FIAT_IN) {
      if (o !== "" && ledgers[oKey] === undefined) {
        ledgers[oKey] = makeFifoLedger(oKey);
      }
      remove(ledgers[oKey], parseFloat(row[H.OUT_AMT]), fiat, date, t);
    }

    // token trade
    if (t === TYPE.BUY || t === TYPE.TRADE) {
      if (i !== "" && ledgers[iKey] === undefined) {
        ledgers[iKey] = makeFifoLedger(iKey);
      }
      if (o !== "" && ledgers[oKey] === undefined) {
        ledgers[oKey] = makeFifoLedger(oKey);
      }

      const inIsXTZ = i === XTZ;
      const outIsXTZ = o === XTZ;
      let inAmt = parseFloat(row[H.IN_AMT]);
      let outAmt = parseFloat(row[H.OUT_AMT]);
      let inFiat = 0;
      let outFiat = 0;
      if (inIsXTZ) {
        inFiat = fiat;
        outFiat = (fiat * inAmt) / outAmt;
      } else if (outIsXTZ) {
        inFiat = (fiat * outAmt) / inAmt;
        outFiat = fiat;
      } else {
        console.error("token-to-token trade or buy not supported yet");
        continue;
      }

      add(ledgers[iKey], inAmt, inFiat, date);
      remove(ledgers[oKey], outAmt, outFiat, date, t);
    }
  }
  const dispossesedLedgers = Object.values(ledgers).filter(
    (l) => l.dispossesed.length > 0
  );

  const allRows = dispossesedLedgers.reduce((acc, l) => {
    return acc.concat(l.dispossesed);
  }, []);

  await csvIO.write(`gains/${year}-gains.csv`, allRows);

  // console.log(dispossesedLedgers.map((l) => l.dispossesed));
};

export default {
  generateReport,
};
